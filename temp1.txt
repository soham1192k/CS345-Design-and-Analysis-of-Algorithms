void solve(){
    int n;cin>>n;
    string s;cin>>s;
    int dp[n][2];
    dp[0][0]=dp[0][1]=0;
    if(s[0]=='X') dp[0][1]=1e9;
    else if(s[0]=='O') dp[0][0]=1e9;
    for(int i=1;i<n;i++){
        if(s[i]=='F'){
            dp[i][0]=min(dp[i-1][1]+1,dp[i-1][0]);
            dp[i][1]=min(dp[i-1][0]+1,dp[i-1][1]);
        }
        else if(s[i]=='X'){
            dp[i][0]=min(dp[i-1][0],dp[i-1][1]+1);
            dp[i][1]=1e9;
        }
        else{
            dp[i][0]=1e9;
            dp[i][1]=min(dp[i-1][1],dp[i-1][0]+1);
        }
    }
    cout<<min(dp[n-1][0],dp[n-1][1])<<'\n';
}
